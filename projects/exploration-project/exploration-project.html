Exploration Project

Paul Hohbaum and Colin Leslie

The purpose of this project was to experiment with a new device and a raspberry pi to try to create something new. We chose to use a 32x128 OLED display.

The program we wrote for the display spawns a square on the screen that will navigate around the display to collect some food circles that are randomly spawned when the program is started.The square stops moving once all of the food has been collected.


Click on the images to enlarge them.

An overview of what the device looks like. The raspberry pi is connected to a breadboard using a cobbler. Four jumper cables are used to connect the cobbler to the OLED display. The display uses the I2C interface.

How the game looks after its collected several peices of food. The square spawns at a random location along with 50-100 peices of food also randomly spawned. The square then navigates to the closest peice of food to consume it.

The square has now collected a few peices of food.

Only one peice of food left. The program is almost done.

Once the square has collected each peice of food it stops moving and the program stops.

This is what the terminal output looks like after the game is complete. On startup each of the locations of the food is displayed. After the square collects a food item the consol prints 'Mhmm yummy'. Once there is no more food it prints 'All done!'.

Challanges

The most difficult part was getting the OLED to display something. Since we didn't have a part number for the OLED it was tough to find a library that was compatible. In the end we used Adafruits Python_SSD1306 library which made drawing shapes much simplier.

'''
IMPORTS
'''
import time
import random

import Adafruit_GPIO.SPI as SPI
import Adafruit_SSD1306

from PIL import Image
from PIL import ImageDraw
from PIL import ImageFont
#--------------------------------------------------------------------------
'''
LIBRARY
(the part that makes the OLED work, credit to the people who wrote the library)
'''
# Raspberry Pi pin configuration:
RST = 24
# Note the following are only used with SPI:
DC = 23
SPI_PORT = 0
SPI_DEVICE = 0

# Beaglebone Black pin configuration:
# RST = 'P9_12'
# Note the following are only used with SPI:
# DC = 'P9_15'
# SPI_PORT = 1
# SPI_DEVICE = 0

# 128x32 display with hardware I2C:
disp = Adafruit_SSD1306.SSD1306_128_32(rst=RST)

# 128x64 display with hardware I2C:
# disp = Adafruit_SSD1306.SSD1306_128_64(rst=RST)

# Note you can change the I2C address by passing an i2c_address parameter like:
# disp = Adafruit_SSD1306.SSD1306_128_64(rst=RST, i2c_address=0x3C)

# Alternatively you can specify an explicit I2C bus number, for example
# with the 128x32 display you would use:
# disp = Adafruit_SSD1306.SSD1306_128_32(rst=RST, i2c_bus=2)

# 128x32 display with hardware SPI:
# disp = Adafruit_SSD1306.SSD1306_128_32(rst=RST, dc=DC, spi=SPI.SpiDev(SPI_PORT, SPI_DEVICE, max_speed_hz=8000000))

# 128x64 display with hardware SPI:
# disp = Adafruit_SSD1306.SSD1306_128_64(rst=RST, dc=DC, spi=SPI.SpiDev(SPI_PORT, SPI_DEVICE, max_speed_hz=8000000))

# Alternatively you can specify a software SPI implementation by providing
# digital GPIO pin numbers for all the required display pins.  For example
# on a Raspberry Pi with the 128x32 display you might use:
# disp = Adafruit_SSD1306.SSD1306_128_32(rst=RST, dc=DC, sclk=18, din=25, cs=22)

# Initialize library.
disp.begin()

# Clear display.
disp.clear()
disp.display()

# Create blank image for drawing.
# Make sure to create image with mode '1' for 1-bit color.
width = disp.width
height = disp.height
image = Image.new('1', (width, height))

# Get drawing object to draw on image.
draw = ImageDraw.Draw(image)
#--------------------------------------------------------------------------
'''
Global Variables
'''
#Random starting co-ordinates
x = random.randint(0,124)
y = random.randint(0,28)

#User defined variables
player_size = 4
grow = False
speed = 1
food_size = 4
num_food = random.randint(50,100)
#--------------------------------------------------------------------------
'''
FUNCTIONS
'''
#Starts the program, creates the food and draws it
def Start():
    food_location = []
    for i in range(num_food):
        food_x = random.randint(0,128-player_size)
        food_y = random.randint(0,32-player_size)
        food = [food_x,food_y]
        food_location.append(food)

    print("Starting position is {},{}".format(x,y))
    print("There is {} food to eat".format(num_food))

    i=1
    for food in food_location:
        food_x = food[0]
        food_y = food[1]
        print("Food {} is at location {},{}".format(i,food_x,food_y))
        i+=1
    return food_location

#Returns the distance to the food
def foodDistance(x,y,food):
    food_x = food[0]
    food_y = food[1]

    dist_x = abs(food_x - x)
    dist_y = abs(food_y - y)

    total_distance = dist_x + dist_y

    return total_distance

#Gets the minimum distance to the next food
def getMinDistance(x,y,food_location):

    min = food_location[0]
    min_distance = foodDistance(x,y,min)

    for food in food_location:
        temp = food
        temp_food_distance = foodDistance(x,y,temp)

        if (temp_food_distance < min_distance):
            min = temp
            min_distance = foodDistance(x,y,min)

    return min


#Moves the player to the x,y coordinate of the food it is chasing
def getFood(x,y,food):
    food_x = food[0]
    food_y = food[1]

    while (food_x != x):
        if (food_x > x):
            x = plusX(x,y)
        else:
            x = minusX(x,y)

    while (food_y != y):
        if (food_y > y):
            y = plusY(x,y)
        else:
            y = minusY(x,y)

    print("Mmmm yummy")
    return x,y

#Draws a peice of food at a given location on the display
def drawFood(food_location):
    for food in food_location:
        food_x = food[0]
        food_y = food[1]
        draw.ellipse((food_x,food_y,food_x+food_size,food_y+food_size), outline=0, fill=1)
#--------------------------------------------------------------------------
'''
MOVEMENT FUNCTIONS

- All of the directions take an input of x,y of the player
- If the player will go out of bounds, the function will return
  the same x or y values that were meant to be changed
'''
#Moves the player to the right (positive x direction)
def plusX(x,y):

    if (x+player_size < width):
        draw.rectangle((x,y,x+player_size,y+player_size), outline=0, fill=20)
        disp.image(image)
        disp.display()
        x+=speed
    return x

#Moves the player to the left (negative x direction)
def minusX(x,y):

    if (x > 0):
        draw.rectangle((x,y,x+player_size,y+player_size), outline=0, fill=1)
        disp.image(image)
        disp.display()
        x-=speed
    return x

#Moves the player down (positive y direction)
def plusY(x,y):

    if (y+player_size < height):
        draw.rectangle((x,y,x+player_size,y+player_size), outline=0, fill=1)
        disp.image(image)
        disp.display()
        y+=speed
    return y

#Moves the player up (negative y direction)
def minusY(x,y):

    if (y > 0):
        draw.rectangle((x,y,x+player_size,y+player_size), outline=0, fill=1)
        disp.image(image)
        disp.display()
        y-=speed
    return y
#--------------------------------------------------------------------------=
'''
MAIN PROGRAM
'''
Done = False
while (Done == False):
    try:

        #TODO implement keyboard controls
        #Direction = up == key w
        #Direction = down == key s
        #Direction = left == key a
        #Direction = right == key d

        #Makes the fod list and draws the food
        food_location = Start()
        drawFood(food_location)

        #Runs until all the food is eaten
        for i in range(num_food):

            #Finds the food that is closest
            min_food = getMinDistance(x,y,food_location)

            #Goes to the closest food
            x,y = getFood(x,y,min_food)

            #Removes that peice of food from the location list and num_food
            food_location.remove(min_food)
            num_food -= 1

            #Grows the player if the option is picked
            if (grow == True):
                player_size+=0.5


        Done = True
        print("All done!")

    except KeyboardInterrupt:
        print("\nGoodbye!")
        exit()
