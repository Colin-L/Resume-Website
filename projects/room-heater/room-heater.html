Automatic Room Heater

For my indoor hydroponics system I was running into some issues with the temperature. My hydroponics system is setup in a room in the basement of my house, as a result it can get quite cold down there if no extra heating is used. If the room temperature drops below 18°C plant growth will slow down significantly. From what I understand the ideal temperature range for plant growth is 22°C to 26°C so I built this device to keep the room within that range.

]The device takes temperature readings from a DHT11 temperature sensor, outputs the reading in celsius to two seven segment displays and if the temperture is below the desiered threshold it will turn on a space heater until the upper threshold is reached.


Click on the images to enlarge them.

Overview of the circuit. The board is powered by a 12V power supply into the arduinos Vin pins. To power all of the segments in the two 7-segment displays I had to use 16 of the arduino nanos I/O pins. After adding the temperature sensor and the relay for the outlet all but 2 of the I/O pins are in use.

Since the arduinos I/O pins have to power so many things the nano can get quite hot. To help cool the arduino I attached a small heatsink to the arduino above the voltage regulator. I probably should have put some resistors between the I/O pins and the segments but I didn't have enough to do the whole board and it the nano only gets to about 60°C with the heat sink attached ¯\_(ツ)_/¯

Side view of the device. The outlet box sits snugly in a wood casing and the circuit is stuck to the wood with velcro tape.

The underside of the circuit. I chose to use veroboard for this project since there was already so many connections, using this type of board saved me from having to add a lot more jumpers.

Top view of the device. The two relay contacts on the top of the circuit board feed into the outlet box where they connect to a 120V 15A relay that allows the arduino to toggle the outlet on and off.

The space heater is used for this project. A sunbeam 1500W space heater. An important detail about this heater is that its controlled by physical switches. This allows me to set the dials to the right settings then turn it on and off by switching the outlet supply.

Serial Monitor of the DHT11ErrCount example program from the SimpleDHT library.This program polls a DHT11 sensor every 1.5 seconds and trys to get a reading. For some reason the DHT11 fails to read a temperature value ~9/10 times and I'm not really sure why. I suspect that the problem is with the nano board that I'm using becuase I've tired mutliple DHT11 sensors from different manufacturers with similar results. To compensate for the inconsistant readings there is some additional error handling code. Check out the code below, try it out for yourself and let me know what you think at colinbleslie@gmail.com

#include <SimpleDHT.h>
#include <stdio.h>

#define MAX_OUT_CHARS 16  //max nbr of characters to be sent on any one serial command

int MAX_TEMP = 26;
int MIN_TEMP = 22;

void setup() {
  Serial.begin(115200);
  pinMode(HEATER_PIN,OUTPUT);

  //Segment 1
  pinMode(A1_SEGMENT,OUTPUT);
  pinMode(B1_SEGMENT,OUTPUT);
  pinMode(C1_SEGMENT,OUTPUT);
  pinMode(D1_SEGMENT,OUTPUT);
  pinMode(E1_SEGMENT,OUTPUT);
  pinMode(F1_SEGMENT,OUTPUT);
  pinMode(G1_SEGMENT,OUTPUT);
  pinMode(DECIMAL1_SEGMENT,OUTPUT);

  //Segment 2
  pinMode(A2_SEGMENT,OUTPUT);
  pinMode(B2_SEGMENT,OUTPUT);
  pinMode(C2_SEGMENT,OUTPUT);
  pinMode(D2_SEGMENT,OUTPUT);
  pinMode(E2_SEGMENT,OUTPUT);
  pinMode(F2_SEGMENT,OUTPUT);
  pinMode(G2_SEGMENT,OUTPUT);
  pinMode(DECIMAL2_SEGMENT,OUTPUT);

  //Tests that everything is working on startup
  //Will turn on  heater for 5 seconds and countdown then turn off
  //and start normal function
  digitalWrite(HEATER_PIN,HIGH);
  Serial.println("Setup:");
  for (int i=9; i>-1; i--){
    displayNumber(i*11);
    delay(500);
  }
  digitalWrite(HEATER_PIN,LOW);
  displayNumber(99);
} //Setup end

void loop() {
  temperature = readTemperature();
  Serial.print("Total reads: "); Serial.print(readCount);
  Serial.print(", err: "); Serial.print(errCount);
  Serial.print(", success rate: "); Serial.print((readCount - errCount) * 100.0 / (float)readCount); Serial.println("%");

  //If the temperature reading succceded display the value, otherwise display the last successful reading
  if (temperature != 999){
    displayValue = temperature;
  }
  displayNumber(displayValue);
  delay(1000);
  clearDisplay(0);
  clearDisplay(1);
  delay(6000);

  if (displayValue >= MAX_TEMP) {
    digitalWrite(HEATER_PIN,LOW);
  }

  if (displayValue <= MIN_TEMP) {
    digitalWrite(HEATER_PIN,HIGH);
  }
}

//----------------------------------------------------Functions---------------------------------------------------
int readTemperature() {
  byte t = 999;
  byte h = 999;
  err = SimpleDHTErrSuccess;
  readCount ++;
  if ((err = dht11.read(TEMP_SENSOR_PIN, &t, &h, NULL)) != SimpleDHTErrSuccess) {
    Serial.print("Read DHT11 failed, err="); Serial.println(err);
    concecutiveFails ++;
    errCount ++;
    //If the reading fails 100 times in a row it will set the temp to 99. This will turn off the heater.
    //This is to prevent the heater from staying on forever in the event that the sensor keeps failing 
    //while the heater is on
    if (concecutiveFails >=100){
      Serial.println("Concecutive Fails: "); Serial.println(concecutiveFails);
      return 99;
    }
    return 999;
  }
  Serial.println("======================================================================");
  Serial.print("Temperature: "); Serial.print(t); Serial.println(" *C ");
  Serial.print("Humidity:    "); Serial.print(h); Serial.println("% ");
  Serial.println("======================================================================");
  concecutiveFails = 0;
  return (int)t;
} //Read temperature end

//Given a 2 or 1 digit number it will display it on the two seven segment displays
//digit[0] Tens digit, digit[1] Ones digit
void displayNumber(int num){
  int digit[2];
  if (num > 9){
    digit[0] = num / 10;
    digit[1] = num % 10;
    digits = 2;
  }else{
    digits = 1;
    digit[0] = num;
  }
  for (int i = 0; i<digits; i++){
    switch(digit[i]){ //Switch on digit[i]: 
      case 1:
        displayOne(i);
        break;
      case 2:
        displayTwo(i);
        break;
      case 3:
        displayThree(i);
        break;
      case 4:
        displayFour(i);
        break;
      case 5:
        displayFive(i);
        break;
      case 6:
        displaySix(i);
        break;
      case 7:
        displaySeven(i);
        break;
      case 8:
        displayEight(i);
        break;
      case 9:
        displayNine(i);
        break;
      case 0:
        displayZero(i);
        break;
      default:
        Serial.println("Default Switch Case Reached Somehow");
  }
}
}
